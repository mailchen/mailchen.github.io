    Data Source=orcl; User ID=demooauser; Password=demooapassword;

    SQL Server 连接串

    Data Source=192.168.11.31; Integrated Security=false; Initial Catalog=gdoa_db; User Id = username; Password = userpass;

    MySQL 连接串

    Data Source=127.0.0.1; Database=china_weather_air; User Id=root; Password=root; SSL Mode=None; charset=utf8;

    Access 连接串

    Provider=Microsoft.Jet.OLEDB.4.0; Data Source=C:\mydatabase.mdb; User Id=用户名; Password=密码;
    Provider=Microsoft.Jet.OLEDB.4.0; Data Source=C:\mydatabase.mdb; Jet OLEDB:Database Password=密码;

    Excel

    Provider=Microsoft.jet.oledb.4.0; Data Source=d:\myExcel.xls; Extended Properties=Excel8.0
